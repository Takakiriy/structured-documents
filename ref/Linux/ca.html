graph TB
    subgraph KEY["🔑 秘密鍵 .key"]
        KEY_DESC["サーバーだけが持つ秘密の鍵
        ・暗号化と復号の核心
        ・絶対に外部に漏らさない
        ・最初に生成する"]
    end

    subgraph CSR["📋 証明書署名要求 .csr"]
        CSR_DESC["私はlocalhostです という申請書
        ・サーバー情報 CN=localhost 等
        ・公開鍵を含む（秘密鍵から導出）
        ・証明書の発行を依頼する書類"]
    end

    subgraph CRT["📜 証明書 .crt"]
        CRT_DESC["この者は本物です という証明書
        ・CAの署名が入っている
        ・SAN, 有効期限, 用途の情報
        ・クライアントに公開する"]
    end

    subgraph CA["🏛️ 認証局 CA"]
        CA_DESC["証明書を発行する信頼された機関
        ・自己署名の場合は自分自身がCA
        ・CA秘密鍵で署名する"]
    end

    KEY -->|"① 秘密鍵からCSRを生成
    openssl req -new"| CSR
    CSR -->|"② CAに提出して署名を依頼"| CA
    CA -->|"③ 署名して証明書を発行
    openssl x509 -req -CA ca.crt"| CRT

    subgraph SELF["💡 自己署名の場合"]
        SELF_DESC["openssl req -new -x509 で
        ①②③を一度にまとめて実行
        CSR不要で直接 .crt を生成
        自分自身がCAになる"]
    end

    KEY -.-> SELF
    SELF -.-> CRT

    subgraph HTTPS["🔒 HTTPS通信"]
        direction LR
        SERVER["サーバー
        .key + .crt を配置"]
        CLIENT["クライアント
        vault CLI / ブラウザ"]
        SERVER <-->|"証明書を提示して暗号化通信"| CLIENT
    end

    KEY -.->|"配置"| SERVER
    CRT -.->|"配置"| SERVER
