flowchart TB
  %% HTML埋め込み向け: 改行は 
 を使用。角括弧は不使用。

  subgraph CI["CI"]
    CISecrets("CIのシークレットに Vault ROLE_ID + SECRET_ID
永続: CIのシークレット管理に保存
注: CIはVaultトークンを保持しない")
    CIDockerExec("docker exec ROLE_ID + SECRET_ID をコンテナーへ一時注入")
  end

  subgraph Container["コンテナー"]
    CInput("一時入力
ROLE_ID + SECRET_ID")
    CLogin("AppRoleでVaultへログイン
ROLE_ID + SECRET_ID を送る")
    CVaultToken("ログインに成功して Vaultトークン取得
短TTL
一時: コンテナー内のみ")
    CReadPAT("VaultのKVからGitLab PATを取得
一時: メモリ上")
    CApprove("git credential approve
GitLab PAT を credential-cache メモリーへ投入")
    CClone("git clone
HTTPSでGitLabへアクセス")
  end

  subgraph Vault["Vault"]
    VAppRole("AppRole設定
ポリシー: 指定パス read のみ
トークンTTL: 短い")
    VKV("KVに保存してある GitLab PAT
永続: username と token")
  end

  subgraph Cache["git-credential-cache"]
    GCache("GitLab PAT をメモリに短時間保持
timeoutで消える")
  end

  subgraph GitLab["GitLab"]
    GL("リポジトリ")
  end

  %% フロー
  CISecrets --> CIDockerExec --> CInput --> CLogin --> VAppRole --> CVaultToken
  CVaultToken --> CReadPAT --> VKV --> CReadPAT
  CReadPAT --> CApprove --> GCache --> CClone --> GL

  %% 色分け
  %% ROLE_ID + SECRET_ID
  classDef rs fill:#E3F2FD,stroke:#1565C0,stroke-width:1px,color:#0D47A1;
  class CISecrets,CIDockerExec,CInput,CLogin rs;

  %% Vaultトークン
  classDef vt fill:#F3E5F5,stroke:#6A1B9A,stroke-width:1px,color:#4A148C;
  class CVaultToken vt;

  %% GitLab PAT
  classDef pat fill:#FFF8E1,stroke:#FF8F00,stroke-width:1px,color:#E65100;
  class VKV,CReadPAT,CApprove pat;

  %% git-credential-cache
  classDef gcc fill:#E8F5E9,stroke:#2E7D32,stroke-width:1px,color:#1B5E20;
  class GCache gcc;
